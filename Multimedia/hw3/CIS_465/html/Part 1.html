def display_img(img, gray):
    '''
    img = input image or image like array
    gray = bool for displaying a grayscale image
    '''
    plt.axis('off')
    if(gray):
        imgplot = plt.imshow(img, cmap='gray', vmin=0, vmax=255)
    else:
        imgplot = plt.imshow(img)
